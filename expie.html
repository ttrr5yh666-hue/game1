// Функция автоматической сборки Expie из отдельных деталек function drawAssembledExpie(x, y) { // Считаем покачивание лап от времени (синусоида) let legSwing = Math.sin(animTime * 0.15) * 10; // Если Expie стоит и не двигается, лапы не качаются if (!keys["ArrowRight"] && !keys["ArrowLeft"]) { legSwing = 0; } // Начинаем рисовать детали сзади наперед, чтобы они правильно перекрывали друг друга // 1. ДЕТАЛЬ: Пушистый хвост (рисуется самым первым, за спиной) if (sprites.tail.complete) { // Хвост немного покачивается в такт ходьбе ctx.drawImage(sprites.tail, x - 20, y + 25 + Math.sin(animTime * 0.1) * 3, 25, 25); } // 2. ДЕТАЛЬ: ЗАДНЯЯ НОГА (Бедро + Голень + Стопа) if (sprites.thigh.complete && sprites.crus.complete && sprites.foot.complete) { let backLegOffset = legSwing; // Качается в одну сторону ctx.drawImage(sprites.thigh, x + 2 + backLegOffset * 0.3, y + 38, 10, 12); ctx.drawImage(sprites.crus, x + 2 + backLegOffset * 0.6, y + 48, 8, 12); ctx.drawImage(sprites.foot, x + backLegOffset, y + 58, 12, 6); } // 3. ДЕТАЛЬ: Нижняя часть туловища (Таз / DownTorso) if (sprites.downTorso.complete) { ctx.drawImage(sprites.downTorso, x + 2, y + 30, 20, 12); } // 4. ДЕТАЛЬ: Верхняя часть туловища (Грудь / UpTorso) if (sprites.upTorso.complete) { ctx.drawImage(sprites.upTorso, x, y + 14, 24, 18); } // 5. ДЕТАЛЬ: Длинная изогнутая шея (крепится к груди) if (sprites.neck.complete) { ctx.drawImage(sprites.neck, x + 6, y, 10, 16); } // 6. ДЕТАЛЬ: Башка шлема (ставится на самый верх шеи) if (sprites.head.complete) { ctx.drawImage(sprites.head, x + 4, y - 10, 20, 14); } // 7. ДЕТАЛЬ: ПЕРЕДНЯЯ НОГА (рисуется поверх туловища) if (sprites.thigh.complete && sprites.crus.complete && sprites.foot.complete) { let frontLegOffset = -legSwing; // Качается в противоположную сторону ctx.drawImage(sprites.thigh, x + 10 + frontLegOffset * 0.3, y + 38, 10, 12); ctx.drawImage(sprites.crus, x + 10 + frontLegOffset * 0.6, y + 48, 8, 12); ctx.drawImage(sprites.foot, x + 8 + frontLegOffset, y + 58, 12, 6); } } function gameLoop() { // Очищаем экран и заливаем темным фоном шахты ctx.clearRect(0, 0, canvas.width, canvas.height); ctx.fillStyle = "#0a0d12"; ctx.fillRect(0, 0, canvas.width, canvas.height); // Рисуем пол под ногами ctx.fillStyle = "#2c3540"; ctx.fillRect(0, 145, canvas.width, 95); // Логика перемещения базы Expie if (keys["ArrowRight"] && player.x < canvas.width - 40) { player.x += player.speed; animTime++; // Крутим счетчик анимации только при ходьбе } if (keys["ArrowLeft"] && player.x > 20) { player.x -= player.speed; animTime++; } // Запускаем модульный конструктор сборки drawAssembledExpie(player.x, player.y); requestAnimationFrame(gameLoop); } // Запуск главного цикла игры gameLoop();
